SELECT k.nev AS FuszerkeverekNev, COUNT(ko.osszetevoid) AS OsszetevokSzama
FROM keverek AS k
INNER JOIN kapcsolat AS ko ON k.id = ko.keverekid
GROUP BY k.nev
ORDER BY OsszetevokSzama DESC
