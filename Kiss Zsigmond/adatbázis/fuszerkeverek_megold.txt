5,
SELECT  nev, Count(osszetevoid) AS [Összetevők száma]
FROM keverek, kapcsolat
WHERE id = keverekid
GROUP BY nev
HAVING COUNT(osszetevoid) = (SELECT MAX(szam) FROM (SELECT COUNT(osszetevoid) AS szam FROM fuszerkeverek GROUP BY nev) AS counts)
ORDER BY osszetevok_szama DESC;
