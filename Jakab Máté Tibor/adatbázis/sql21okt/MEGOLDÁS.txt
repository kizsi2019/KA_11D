CREATE TABLE alloviz (
    id INT PRIMARY KEY,
    nev NVARCHAR(255),
    tipus NVARCHAR(255),
    terulet FLOAT,
    vizgyujto FLOAT
);

CREATE TABLE helykapcs (
    allovizid INT,
    gpsid INT,
    PRIMARY KEY (allovizid, gpsid)
);

CREATE TABLE telepulesgps (
    id INT PRIMARY KEY,
    nev NVARCHAR(255),
    hosszusag DECIMAL(8,2),
    szelesseg DECIMAL(8,2)
);
